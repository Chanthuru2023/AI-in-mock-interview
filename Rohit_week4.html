from PIL import Image

# Function to verify images and remove corrupted ones
def verify_images(directory):
    if not os.path.exists(directory):
        print(f"Directory does not exist: {directory}")
        return
    
    total_images = 0
    corrupted_images = 0

    for emotion in os.listdir(directory):
        emotion_path = os.path.join(directory, emotion)
        
        if not os.path.isdir(emotion_path):
            continue
        
        for img_file in os.listdir(emotion_path):
            img_path = os.path.join(emotion_path, img_file)
            
            if not img_file.lower().endswith(('.png', '.jpg', '.jpeg', '.bmp', '.gif', '.tiff')):
                continue
            
            total_images += 1
            
            try:
                img = Image.open(img_path)
                img.verify()  # Raise an exception if corrupted
                
            except (IOError, SyntaxError):
                os.remove(img_path)  # Remove corrupted file
                corrupted_images += 1

    print(f"\nSummary for directory: {directory}")
    print(f"Total images processed: {total_images}")
    print(f"Corrupted images removed: {corrupted_images}")

# Clean both train and test datasets
verify_images(train_dir)
verify_images(test_dir)

# Function to display a few sample images from the generator
def display_sample_images(generator, num_images=5):
    plt.figure(figsize=(10, 10))
    for i in range(num_images):
        img, label = next(generator)  # Get the next batch of images and labels
        plt.subplot(1, num_images, i + 1)
        plt.imshow(img[0], cmap='gray')  # Display the first image of the batch
        plt.title(f"Label: {label[0].argmax()}")  # Show the label of the image
        plt.axis('off')
    plt.show()

# Display sample images from the training generator

import matplotlib.pyplot as plt

# Function to plot class distribution
def plot_class_distribution(class_distribution, title, save_as=None):
    if not class_distribution:
        print("Warning: The class distribution is empty.")
        return
    
    classes = list(class_distribution.keys())
    counts = list(class_distribution.values())
    
    plt.figure(figsize=(10, 5))
    
    # Map each class to its corresponding color
    color_map = {  # Define colors for each emotion
        'happy': 'yellow',    # Color for happy
        'sad': 'blue',        # Color for sad
        'angry': 'red',       # Color for angry
        'surprised': 'orange', # Color for surprised
        'fearful': 'purple',  # Color for fearful
        'disgusted': 'green',  # Color for disgusted
        'neutral': 'gray'     # Color for neutral
    }
    
    # Use the color from the map
    bar_colors = [color_map.get(class_name, 'black') for class_name in classes]  # Default to black if not found
    
    plt.bar(classes, counts, color=bar_colors)
    plt.title(title)
    plt.xlabel('Emotion Classes')
    plt.ylabel('Number of Images')
    plt.xticks(rotation=45)
    
    if save_as:
        plt.savefig(save_as, bbox_inches='tight')
    plt.show()
